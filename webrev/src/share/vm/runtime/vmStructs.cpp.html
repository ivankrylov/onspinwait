3296 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3297   {
3298     VMTypeEntry* types = origtypes;
3299     while (types->typeName != NULL) {
3300       if (strcmp(typeName, types->typeName) == 0) {
3301         // Found it
3302         return 1;
3303       }
3304       ++types;
3305     }
3306   }
3307   // Search for the base type by peeling off const and *
3308   size_t len = strlen(typeName);
3309   if (typeName[len-1] == '*') {
3310     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3311     strncpy(s, typeName, len - 1);
3312     s[len-1] = '\0';
3313     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3314     if (recursiveFindType(origtypes, s, true) == 1) {
3315       FREE_C_HEAP_ARRAY(char, s);
3316       return 1;
3317     }
3318     FREE_C_HEAP_ARRAY(char, s);
3319   }
3320   const char* start = NULL;
3321   if (strstr(typeName, "GrowableArray<") == typeName) {
3322     start = typeName + strlen("GrowableArray<");
3323   } else if (strstr(typeName, "Array<") == typeName) {
3324     start = typeName + strlen("Array<");
3325   }
3326   if (start != NULL) {
3327     const char * end = strrchr(typeName, '>');
3328     int len = end - start + 1;
3329     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3330     strncpy(s, start, len - 1);
3331     s[len-1] = '\0';
3332     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3333     if (recursiveFindType(origtypes, s, true) == 1) {
3334       FREE_C_HEAP_ARRAY(char, s);
3335       return 1;
3336     }
3337     FREE_C_HEAP_ARRAY(char, s);
3338   }
3339   if (strstr(typeName, "const ") == typeName) {
3340     const char * s = typeName + strlen("const ");
3341     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3342     if (recursiveFindType(origtypes, s, true) == 1) {
3343       return 1;
3344     }
3345   }
3346   if (strstr(typeName, " const") == typeName + len - 6) {
3347     char * s = os::strdup_check_oom(typeName);
3348     s[len - 6] = '\0';
3349     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3350     if (recursiveFindType(origtypes, s, true) == 1) {
3351       os::free(s);
3352       return 1;
3353     }
3354     os::free(s);
3355   }
3356   if (!isRecurse) {
3357     tty->print_cr("type \"%s\" not found", typeName);
3358   }
3359   return 0;
3360 }
