3315 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3316   {
3317     VMTypeEntry* types = origtypes;
3318     while (types->typeName != NULL) {
3319       if (strcmp(typeName, types->typeName) == 0) {
3320         // Found it
3321         return 1;
3322       }
3323       ++types;
3324     }
3325   }
3326   // Search for the base type by peeling off const and *
3327   size_t len = strlen(typeName);
3328   if (typeName[len-1] == '*') {
3329     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3330     strncpy(s, typeName, len - 1);
3331     s[len-1] = '\0';
3332     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3333     if (recursiveFindType(origtypes, s, true) == 1) {
3334       FREE_C_HEAP_ARRAY(char, s);
3335       return 1;
3336     }
3337     FREE_C_HEAP_ARRAY(char, s);
3338   }
3339   const char* start = NULL;
3340   if (strstr(typeName, "GrowableArray<") == typeName) {
3341     start = typeName + strlen("GrowableArray<");
3342   } else if (strstr(typeName, "Array<") == typeName) {
3343     start = typeName + strlen("Array<");
3344   }
3345   if (start != NULL) {
3346     const char * end = strrchr(typeName, '>');
3347     int len = end - start + 1;
3348     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3349     strncpy(s, start, len - 1);
3350     s[len-1] = '\0';
3351     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3352     if (recursiveFindType(origtypes, s, true) == 1) {
3353       FREE_C_HEAP_ARRAY(char, s);
3354       return 1;
3355     }
3356     FREE_C_HEAP_ARRAY(char, s);
3357   }
3358   if (strstr(typeName, "const ") == typeName) {
3359     const char * s = typeName + strlen("const ");
3360     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3361     if (recursiveFindType(origtypes, s, true) == 1) {
3362       return 1;
3363     }
3364   }
3365   if (strstr(typeName, " const") == typeName + len - 6) {
3366     char * s = os::strdup_check_oom(typeName);
